      return res.status(404).json({
        status: "error",
        message: "No bookmarks found",
      });
    }
    res.status(200).json({
      status: "success",
      data: {
        bookmarks,
      },
    });
  } catch (err) {
    res.status(400).json({
      status: "error",
      message: err.message,
    });
  }
};
